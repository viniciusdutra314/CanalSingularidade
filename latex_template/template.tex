\newcommand{\coded}[5]{
    \definecolor{Gray}{gray}{0.9}
    \begin{minipage}{\textwidth}
        \captionof{listing}{{#5}}
        \inputminted[
            frame=lines,
            framesep=2mm,
            baselinestretch=1.2,
            firstline={#3},
            lastline={#4},
            bgcolor=Gray,
            linenos]{#1}{#2}
    \end{minipage}
    %#1: The programming language of the code snippet (e.g., python, java, c++, etc.).
    %2: The filename of the code file to be included.
    %3: The first line number of the code snippet to be included.
    %4: The last line number of the code snippet to be included.
    %5: The caption to be displayed above the code snippet
}
